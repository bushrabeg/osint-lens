**Degisiklik 1 — Yeni fonksiyon: `buildLiveIntelPromptSection()`**

`// ===================== GROQ ANALİZİ =====================` yorumunun hemen ONCESINE bu fonksiyonu ekle:

```javascript
// ===================== GROQ — CANLI VERİ PROMPT BUILDER =====================
function buildLiveIntelPromptSection() {
  const d = window.liveIntelData;
  if (!d) return '';

  function formatItems(items, keyFn) {
    if (!items || items.length === 0) return null;
    return items.slice(0, 5).map(item => {
      const key = keyFn(item);
      const date = item.date || '';
      return '- ' + item.title + ' | ' + key + ' | ' + date;
    }).join('\n');
  }

  const sections = [];

  const rt = formatItems(d.redditTarget, item => 'r/' + item.subreddit);
  if (rt) sections.push('[Reddit — Hedef Araması]\n' + rt);

  const rs = formatItems(d.redditSubreddits, item => 'r/' + item.subreddit);
  if (rs) sections.push('[Reddit — OSINT Subredditler]\n' + rs);

  const rss = formatItems(d.rssFeeds, item => item.source || 'RSS');
  if (rss) sections.push('[RSS — Reuters / BBC / Al Jazeera]\n' + rss);

  const gn = formatItems(d.googleNews, item => item.source || 'Google News');
  if (gn) sections.push('[Google News]\n' + gn);

  if (sections.length === 0) return '';

  return '\n\n== CANLI VERİ ==\nAnalizi yaparken aşağıdaki güncel canlı verileri de göz önünde bulundur:\n\n' + sections.join('\n\n');
}
```
