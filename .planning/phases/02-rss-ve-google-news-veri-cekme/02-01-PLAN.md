    ```javascript
    async function fetchRSSFeeds() {
      window.liveIntelData.loading.rssFeeds = true;
      window.liveIntelData.rssFeeds = [];
      const RSS_SOURCES = [
        { name: 'Reuters', url: 'https://feeds.reuters.com/reuters/topNews' },
        { name: 'BBC', url: 'http://feeds.bbci.co.uk/news/world/rss.xml' },
        { name: 'Al Jazeera', url: 'https://www.aljazeera.com/xml/rss/all.xml' }
      ];
      try {
        const fetches = RSS_SOURCES.map(src => {
          const proxyUrl = 'https://api.allorigins.win/get?url=' + encodeURIComponent(src.url);
          return fetch(proxyUrl)
            .then(r => r.ok ? r.json() : Promise.reject('HTTP ' + r.status))
            .then(data => {
              if (!data.contents) return [];
              const xml = new DOMParser().parseFromString(data.contents, 'text/xml');
              const items = Array.from(xml.querySelectorAll('item'));
              return items.slice(0, 10).map(item => {
                const rawDate = item.querySelector('pubDate') ? item.querySelector('pubDate').textContent : '';
                let date = '';
                try { date = rawDate ? new Date(rawDate).toISOString().split('T')[0] : ''; } catch(e) { date = ''; }
                return {
                  title: item.querySelector('title') ? item.querySelector('title').textContent.trim() : '',
                  source: src.name,
                  date: date,
                  url: item.querySelector('link') ? item.querySelector('link').textContent.trim() : ''
                };
              }).filter(item => item.title && item.url);
            })
            .catch(e => { console.warn('[OSINT Lens] ' + src.name + ' RSS çekilemedi:', e); return []; });
        });
        const results = await Promise.all(fetches);
        window.liveIntelData.rssFeeds = results.flat();
      } catch(e) {
        console.warn('[OSINT Lens] RSS çekme başarısız:', e.message);
        window.liveIntelData.rssFeeds = [];
      } finally {
        window.liveIntelData.loading.rssFeeds = false;
      }
    }

    async function fetchGoogleNews(target) {
      window.liveIntelData.loading.googleNews = true;
      window.liveIntelData.googleNews = [];
      try {
        const gnUrl = 'https://news.google.com/rss/search?q=' + encodeURIComponent(target) + '&hl=en-US&gl=US&ceid=US:en';
        const proxyUrl = 'https://api.allorigins.win/get?url=' + encodeURIComponent(gnUrl);
        const res = await fetch(proxyUrl);
        if (!res.ok) throw new Error('HTTP ' + res.status);
        const data = await res.json();
        if (!data.contents) throw new Error('Bos yanit');
        const xml = new DOMParser().parseFromString(data.contents, 'text/xml');
        const items = Array.from(xml.querySelectorAll('item'));
        window.liveIntelData.googleNews = items.slice(0, 15).map(item => {
          const rawDate = item.querySelector('pubDate') ? item.querySelector('pubDate').textContent : '';
          let date = '';
          try { date = rawDate ? new Date(rawDate).toISOString().split('T')[0] : ''; } catch(e) { date = ''; }
          const sourceEl = item.querySelector('source');
          return {
            title: item.querySelector('title') ? item.querySelector('title').textContent.trim() : '',
            source: sourceEl ? sourceEl.textContent.trim() : 'Google News',
            date: date,
            url: item.querySelector('link') ? item.querySelector('link').textContent.trim() : ''
          };
        }).filter(item => item.title && item.url);
      } catch(e) {
        console.warn('[OSINT Lens] Google News çekilemedi:', e.message);
        window.liveIntelData.googleNews = [];
      } finally {
        window.liveIntelData.loading.googleNews = false;
      }
    }
    ```

    Key constraints:
    - `allorigins.win/get?url=` returns `{ contents: string, status: {...} }` — use `data.contents` field
    - `DOMParser` is built-in browser API, no import needed
    - BBC RSS uses `http://` URL — allorigins proxy handles this, do not change to `https://`
    - `<link>` in RSS XML is text content (between tags), not an attribute — use `.textContent`, not `.getAttribute`
    - Slice items to 10 (RSS) and 15 (Google News) to keep payload manageable
    - Filter out items missing title or url to avoid broken entries in Phase 3 panel
