    function updateSourceBadge(badgeId, state) {
      const badge = document.getElementById(badgeId);
      if (!badge) return;
      badge.className = 'live-intel-source-badge ' + state;
      const labels = { loading: 'YÜKLENİYOR', ok: 'TAMAM', error: 'HATA' };
      badge.textContent = labels[state] || state;
    }

    function renderLiveIntel() {
      const container = document.getElementById('liveIntelSources');
      if (!container) return;

      const d = window.liveIntelData;

      const sources = [
        {
          id: 'src-reddit-target',
          badgeId: 'badge-reddit-target',
          name: 'Reddit — Hedef Araması',
          type: 'reddit',
          items: d.redditTarget,
          loading: d.loading.redditTarget,
          getLabel: item => item.subreddit ? 'r/' + item.subreddit : 'Reddit'
        },
        {
          id: 'src-reddit-subs',
          badgeId: 'badge-reddit-subs',
          name: 'Reddit — OSINT Subredditler',
          type: 'reddit',
          items: d.redditSubreddits,
          loading: d.loading.redditSubreddits,
          getLabel: item => item.subreddit ? 'r/' + item.subreddit : 'Reddit'
        },
        {
          id: 'src-rss',
          badgeId: 'badge-rss',
          name: 'RSS — Reuters / BBC / Al Jazeera',
          type: 'rss',
          items: d.rssFeeds,
          loading: d.loading.rssFeeds,
          getLabel: item => item.source || 'RSS'
        },
        {
          id: 'src-gnews',
          badgeId: 'badge-gnews',
          name: 'Google News',
          type: 'rss',
          items: d.googleNews,
          loading: d.loading.googleNews,
          getLabel: item => item.source || 'Google News'
        }
      ];

      container.innerHTML = sources.map(src => {
        const state = src.loading ? 'loading' : (src.items.length > 0 ? 'ok' : 'error');
        const itemsHtml = src.items.length > 0
          ? '<ul class="live-intel-items">' + src.items.map(item => {
              const label = src.getLabel(item);
              const date = item.date || '';
              const safeTitle = item.title
                ? item.title.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;')
                : '(Başlık yok)';
              const safeUrl = item.url ? item.url.replace(/"/g,'&quot;') : '#';
              return '<li class="live-intel-item">' +
                '<a href="' + safeUrl + '" target="_blank" rel="noopener noreferrer">' + safeTitle + '</a>' +
                '<span class="live-intel-item-meta">' + label + (date ? '  ·  ' + date : '') + '</span>' +
                '</li>';
            }).join('') + '</ul>'
          : '<div class="live-intel-empty">' + (src.loading ? 'Yükleniyor...' : 'Sonuç bulunamadı.') + '</div>';

        return '<div class="live-intel-source-block" id="' + src.id + '" data-type="' + src.type + '">' +
          '<div class="live-intel-source-header">' +
          '<span class="live-intel-source-name">' + src.name + '</span>' +
          '<span class="live-intel-source-badge ' + state + '" id="' + src.badgeId + '">' +
          (state === 'loading' ? 'YÜKLENİYOR' : state === 'ok' ? 'TAMAM' : 'HATA') +
          '</span>' +
          '</div>' +
          itemsHtml +
          '</div>';
      }).join('');

      // Aktif filtreyi koru
      const activeFilter = document.querySelector('.live-intel-filter-btn.active');
      if (activeFilter) {
        const filterVal = activeFilter.getAttribute('onclick').match(/'([^']+)'\s*\)$/);
        if (filterVal && filterVal[1] !== 'all') applyLiveIntelFilter(filterVal[1]);
      }
    }
    ```

    Kritik kisitlar:
    - innerHTML kullanilir cunku DOMPurify bu projede yok (v2 gereksinimi olarak ertelenmiş, REQUIREMENTS.md SEC-01) — title ve url manuel olarak escape edilir
    - `item.title` icin `&`, `<`, `>`, `"` karakterleri replace zinciriyle escape edilir
    - `item.url` icin `"` escape edilir (href attribute guvenligi icin)
    - data-type attribute 'reddit' veya 'rss' degerlerinden birini alir — setLiveIntelFilter() bu degeri kullanir

    function applyLiveIntelFilter(filterVal) {
      const blocks = document.querySelectorAll('#liveIntelSources .live-intel-source-block');
      blocks.forEach(block => {
        if (filterVal === 'all') {
          block.style.display = 'block';
        } else {
          block.style.display = block.getAttribute('data-type') === filterVal ? 'block' : 'none';
        }
      });
    }

    function setLiveIntelFilter(btn, filterVal) {
      document.querySelectorAll('.live-intel-filter-btn').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      applyLiveIntelFilter(filterVal);
    }
    ```

    **Adım 2 — refreshLiveIntel() ekle:**

    `function setLiveIntelFilter()` taninmından SONRA asagidaki fonksiyonu ekle:

    ```javascript
    function refreshLiveIntel() {
      const btn = document.getElementById('liveIntelRefreshBtn');
      if (btn) btn.disabled = true;
      const q = window.liveIntelData.lastTarget;
      if (!q) { if (btn) btn.disabled = false; return; }

      fetchRedditTarget(q).then(() => {
        console.log('[OSINT Lens] [Yenile] Reddit hedef:', window.liveIntelData.redditTarget);
        renderLiveIntel();
      });
      fetchRedditSubreddits().then(() => {
        console.log('[OSINT Lens] [Yenile] Subredditler:', window.liveIntelData.redditSubreddits);
        renderLiveIntel();
      });
      fetchRSSFeeds().then(() => {
        console.log('[OSINT Lens] [Yenile] RSS:', window.liveIntelData.rssFeeds);
        renderLiveIntel();
      });
      fetchGoogleNews(q).then(() => {
        console.log('[OSINT Lens] [Yenile] Google News:', window.liveIntelData.googleNews);
        renderLiveIntel();
        if (btn) btn.disabled = false;
      });
    }
    ```
